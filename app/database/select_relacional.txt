select usuario.usuario_cedula , usuario.usuario_apellido , usuario.usuario_nombre , categoria.categoria_nombre , departamento.departamento_nombre , rol.rol from usuario , usuario_departamento , departamento , categoria , usuario_rol ,rol where usuario_departamento.fk_departamento = departamento.id_departamento and usuario_departamento.fk_usuario = usuario.id_usuario and usuario.fk_categoria = categoria.id_categoria and usuario_rol.fk_rol = rol.id_rol and usuario_rol.fk_usuario = usuario.id_usuario
